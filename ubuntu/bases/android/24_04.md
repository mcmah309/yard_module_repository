```Dockerfile
### Android
# https://github.com/thyrlian/AndroidSDK
FROM thyrlian/android-sdk:10.0

ENV HOME ${HOME:-/root}

RUN apt-get update -y \
    && apt-get upgrade -y \
    && apt-get install -y --no-install-recommends --no-install-suggests \
    ca-certificates \
    && update-ca-certificates \
    && rm -rf /var/lib/apt/lists/*
```
